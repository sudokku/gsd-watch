        var itemLines []string
        for j, part := range nameParts {
            rawText := part
            var text string
            switch {
            case i == t.cursor:

            case isDimmed:

                if isDimmed {
                    cont = tui.PendingStyle.Render(continuation)
                }
                l = cont + text
            }
            itemLines = append(itemLines, l)
        }
        lines = append(lines, strings.Join(itemLines, "\n"))
